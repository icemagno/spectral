CREATE TABLE awk_out
(
  index_id serial NOT NULL,
  id_experiment integer,
  id_activity integer,
  id_pipeline integer,
  eigsolveoption character varying(250),
  optifunc character varying(250),
  caixa1 character varying(250),
  gorder character varying(250),
  mindegree character varying(250),
  maxdegree character varying(250),
  trianglefree character varying(250),
  allowdiscgraphs character varying(250),
  biptonly character varying(250),
  g6file character varying(250),
  g6fileid character varying(250),
  g6splitedfile integer,
  CONSTRAINT awk_out_pkey PRIMARY KEY (index_id),
  CONSTRAINT awk_out_fkac FOREIGN KEY (id_activity)
      REFERENCES activities (id_activity) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT awk_out_fkex FOREIGN KEY (id_experiment)
      REFERENCES experiments (id_experiment) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT awk_out_fkpi FOREIGN KEY (id_pipeline)
      REFERENCES pipelines (id_pipeline) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT g6splitedfile_fk FOREIGN KEY (g6splitedfile)
      REFERENCES files (id_file) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION
)
