CREATE TABLE filter_out
(
  index_id serial NOT NULL,
  id_experiment integer,
  id_activity integer,
  id_instance integer,
  gvfile integer,
  optifunc character varying(250),
  caixa1 character varying(250),
  evaluatedvalue character varying(250),
  g6fileid character varying(250),
  maxresults character varying(250),
CONSTRAINT filter_out_pkey PRIMARY KEY (index_id),
  CONSTRAINT gvfile_fkfile FOREIGN KEY (gvfile)
      REFERENCES files (id_file) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT filter_out_fkac FOREIGN KEY (id_activity)
      REFERENCES activities (id_activity) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT filter_out_fkex FOREIGN KEY (id_experiment)
      REFERENCES experiments (id_experiment) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT filter_out_fkpi FOREIGN KEY (id_instance)
      REFERENCES instances (id_instance) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION
)
