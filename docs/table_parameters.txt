CREATE TABLE spectral_parameters
(
  index_id serial NOT NULL,
  id_experiment integer,
  id_activity integer,
  id_instance integer,
  adjacency character varying(250),
  laplacian character varying(250),
  slaplacian character varying(250),
  optifunc character varying(250),
  caixa1 character varying(250),
  gorder character varying(250),
  mindegree character varying(250),
  maxdegree character varying(250),
  trianglefree character varying(250),
  allowdiscgraphs character varying(250),
  biptonly character varying(250),
  maxresults character varying(250),
  CONSTRAINT spectral_parameters_pkey PRIMARY KEY (index_id),
  CONSTRAINT spectral_parameters_fkac FOREIGN KEY (id_activity)
      REFERENCES activities (id_activity) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT spectral_parameters_fkex FOREIGN KEY (id_experiment)
      REFERENCES experiments (id_experiment) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT spectral_parameters_fkpi FOREIGN KEY (id_instance)
      REFERENCES pipelines (id_instance) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION
)
